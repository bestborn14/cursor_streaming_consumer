# Installation
brew install rbenv
vim ~/.bash_profile
eval "$(rbenv init -)"
source ~/.bash_profile
rbenv install -l
## Install from the list
rbenv install 3.3.3
rbenv global 3.3.3
## Gems
gem install bundler
gem install rails -v 7.1.3.4
